      Простая программка в Builder C++ 2007 на использование массива:

      //#include <vcl.h>
      #include <iostream>
      using namespace std;
      int main(int argc, char* argv[])
      {
      cout<<"obrabotka massiva"<<endl;
      int mas[16]={5,-12,-12,9,10,0,-9,-12,-1,23,65,64,11,43,39,-15}; // объявление и инициализация массива
      cout<<"indeks"<<"\t"<<"element massiva"<<endl;
      for (int i = 0; i < 16; i++) //начало цикла
      {
      //вывод на экран индекса ячейки массива а затем содержимого этой ячейки то есть число
      //ну а escape-последовательность \t вам должна быть известна - табуляция
      cout<<"mas["<<i<<"]"<<"\t\t"<<mas[i]<<endl;
      }
      system("pause");
      return 0;
      }
