Вопросы тестирования по теме:
Функции. Классы памяти и область их действия.

  1. Укажите ключевые слова, используемые для обозначения класса памяти
  2. Класс памяти
  3. Внешние переменные
  4. Внешние статические переменные
  5. Локальные переменные
  6. Статические переменные
  7. Автоматические переменные
  8. Регистровые переменные
  9. Переменные, известные только одной функции, именно той, которая их содержит -
  10. Переменные известные нескольким функциям -

Что будет выведено на экран после исполнения фрагмента программы:

  1. 	int x = 5;
    	void main()
    	{	int x = 2, y = 1; cout << x << ' ' << y << endl; }
  2. 	int x = 5;
    	void main()
    	{	int x = 2, y = 1; 
    		cout << ::x << ' ' << y << endl;
    	}
  3. 	int x = 5;
    	void main()
    	{	int x = 2, y = 1; y += ++::x;
    		cout << ::x << ' ' << y << endl;
    	}
  4. 	int x = 5;
    	mim(int x) { return --::x; }
    	void main()
    	{	int x = 2, y = 1; static int z = 2; --z;
    		mim(x); y = ::x - z; cout << ::x << ' ' << y << endl;
    	}
  5. 	int x = 5;
    	mim(int x) { return --::x; }
    	void main()
    	{	int x = 2, y = 1; static int z = 2; --z;
    		mim(x); y = ::x - z; cout << ::x << ' ' << z << endl;
    	}
