Математические функции и выражения.

Какое значение примет z после исполнения фрагмента программы:

Пояснения:

  • При вводе с клавиатуры значения правильного ответа необходимо ввести только числовое значение.

  • Команда cout << "z = " << z << endl; означает: вывести на экран значение z.

 

  1. 	float a = 2.41, z; z = ceil(a); cout << "z = " << z << endl;
  2. 	float a = -3.11, z; z = ceil(a); cout << "z = " << z << endl;
  3. 	float a = 2.41, z; z = floor(a); cout << "z = " << z << endl;
  4. 	float a = -3.11, z; z = floor(a); cout << "z = " << z << endl;
  5. 	float a = -3.11, z; z = fabs(a); cout << "z = " << z << endl;
  6. 	float a = 3.11, b = 2.41, z; z = fmod(a, b); cout << "z = " << z << endl;
  7. 	int a = 3, z; z = pow10(a); cout << "z = " << z << endl;
  8. 	int a = 4, b = 3, z; z = pow(a, b); cout << "z = " << z << endl;
  9. 	int a = 4, b = 3, z; z = hypot(a, b); cout << "z = " << z << endl;
  10. 	int a = 4, z; z = sqrt(a); cout << "z = " << z << endl;
  11. 	int a = 3; double b = .75, z; z = ldexp(b, a); cout << "z = " << z << endl;
