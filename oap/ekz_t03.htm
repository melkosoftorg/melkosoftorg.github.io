В описании

	struct pin
	{
		int x, y;
		char ch, str;
	};
	struct - это
	pin - это
	int x, y; char ch, str; - это

	union pin
	{
		int x, y;
		char ch, str;
	};
	union - это
	pin - это
	char ch, str; - это
