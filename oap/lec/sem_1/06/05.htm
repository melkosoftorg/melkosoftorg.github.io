Для вычисления значения функции

при значениях аргумента x = 3.74, 3.95, 4.08, 4.2, 4.44 можно предложить программу:

Поскольку выражение громоздкое, при вычислении разобъем его на логически завершенные блоки - числитель и знаменатель.
Значения аргумента x введем с клавиатуры.
Для вывода результатов на экран создадим макроподстановку WW. 

	#include<conio.h>
	#include<stdio.h>
	#include<math.h>
	#define WW printf("y(%5.2f) = %7.3f\n", x, y)
	void main()
	{
		// Объявить переменные
		int i;
		float a, b, x, y;
		// очистить экран
		clrscr();
		// повторить вычисления 5 раз
		for (i = 1; i < 6; i++)
		{
			// ввести значения
			printf("x = "); scanf("%f", &x);
			// вычислить числитель a, знаменатель b, функцию y
			a = sin(x) + sin(M_PI * x * x) + exp(cos(x)) + x * x + log(x * x);
			b = sqrt(sin(M_PI * x * x) + log(x * x));
			y = a / b; 
			// вывести результаты на экран
			WW;
		}
		getch();
	}

	Результаты
			y(3.74) = 10.210
			y(3.95) = 12.809
			y(4.08) = 10.439
			y(4.20) = 13.800
			y(4.44) = 14.645
