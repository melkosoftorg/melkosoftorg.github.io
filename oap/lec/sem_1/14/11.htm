Копирование содержимого одного файла в другой

В программе показаны возможности копирования содержимого одного файла в другой или на принтер. 
	#include<stdio.h>
	#include<string.h>
	#include<stdlib.h>
	#include<dir.h>
	#include<conio.h>
	void main()
	{
		FILE *fp1, *fp2, *ptr;
		char *name1 = "mim.txt";
		char *name2 = "out_file.txt";
		int num; clrscr();
		if ((fp1 = fopen(name1, "r")) == NULL)	// Открыть файл для чтения
		{ 
			puts("Невозможно открыть файл"); exit(1); 
		}
		printf("%s\n\n", searchpath(name1));	// Маршрут доступа к файлу
		/* Открыть файл для вывода на принтер - принтер должен быть 
		    подключен
		    if ((ptr = fopen("PRN", "w")) == NULL)
		    { 
			puts("Принтер не готов к работе"); fclose(fp1); 
			exit(1); 
		    }
		*/
		if ((fp2 = fopen(name2, "w")) == NULL) // Открыть файл для записи
		{ 
			puts("Невозможно открыть файл"); exit(1); 
		}
		printf("%s\n\n", searchpath(name2)); // Маршрут доступа к файлу
		while((num = fgetc(fp1)) != EOF)
		{
			fputc(num, fp2);	// Запись в файл
			printf("%c",num);	// Вывод на экран
			/* fprintf(ptr,"%c",num);// Вывод на принтер */
		}
		fclose(fp1); fclose(fp2); getch();
	}
