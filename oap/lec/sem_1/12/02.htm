Титульный лист фирменного бланка

В программе используются две функции

  • void space(int num)   - вывод num пробелов
  • void starbar() - вывод строки из LIM символов *

Размещение определения функций до функции main() позволяет их не описывать. 

	#include<stdio.h>
	#include<conio.h>
	#include<string.h>	// для функции strlen() - подсчитать количество символов
	#define NAME "Ставропольский колледж связи имени В.А. Петрова"
	#define ADD "пер. Черняховского, 3 тел. 25 - 19 - 78"
	#define PL "факс ..."
	#define LIM 79
	
	// определение функции space - вывод строки из num пробелов
	// получает в качестве параметра количество пробелов, 
	// ничего не возвращает
	void space(int num)
	{
		int col;
		for (col = 1; col <= num; col++)
			putchar(' ');
	}
	
	// определение функции starbar - вывод строки из LIM символов *
	// функция ничего не получает и не возвращает
	void starbar()
	{
		int col;
		for (col = 1; col <= LIM; col++)
			putchar('*');
		putchar('\n');
	}
	
	void main()
	{
		clrscr();
		int sp;

		// первая выводимая строка
		starbar();					// вывести строку из 79 символов *

		// вторая выводимая строка
		space(25);					// вывести 25 пробелов - аргумент функции константа
		printf("%s\n", NAME);			// затем вывести текст

		// третья выводимая строка
		// вычисление количества пробелов для пропуска с начала строки, чтобы
		// следующую строку текста отцентровать при выводе на экран автоматически
		sp = (LIM - strlen(ADD)) / 2; 
		space(sp);					// sp пробелов - аргумент функции переменная
		printf("%s\n", ADD);			// затем вывести текст

		// четвертая выводимая строка
		space((LIM - strlen(PL)) / 2);	// аргумент функции - выражение
		printf("%s\n", PL);

		// пятая выводимая строка
		starbar(); 
		getch();
	}
