Этап 4.   Реализация алгоритма на входном языке ЭВМ

Уже при проектировании алгоритма, возможно, был выбран определенный алгоритмический язык, теперь необходимо осуществить непосредственно программирование.

Программа на языке C++, соответствующая разработанному алгоритму, включает в себя:

  • стандартные библиотеки, подключаемые посредством директивы #include,
  • объявление переменных (как аргументов, так и результатов),
  • ввод значений сторон треугольника,
  • проверку условий на возможность существования треугольника с заданными сторонами,
  • вычисление периметра, площади, высот,
  • вывод результатов вычислений на экран. 
   
	#include<stdio.h>
	#include<conio.h>
	#include<math.h>
	void main(void)
	{
		float a, b, c, p, P, s, ha, hb, hc;
		clrscr();
		gotoxy(10, 4); printf("float a = "); scanf("%f", &a);
		gotoxy(10, 5); printf("float b = "); scanf("%f", &b);
		gotoxy(10, 6); printf("float c = "); scanf("%f", &c);
		if((a + b > c) && (b + c > a) && (a + c > b))
		{	P = a + b + c;
			p = P / 2;
			s = sqrt(p * (p - a) * (p - b) * (p - c));
			ha = 2 * s / a;
			hb = 2 * s / b;
			hc = 2 * s / c;
			gotoxy(10,  8); printf("P = %.2f", P);
			gotoxy(10,  9); printf("s = %.2f", s);
			gotoxy(10, 10); printf("ha = %.2f\thb = %.2f\thc = %.2f", ha, hb, hc);
		}
		else
		{
			gotoxy(10, 8); printf("Недопустимые параметры треугольника");
		} getch();
	}
