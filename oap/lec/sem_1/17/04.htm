Использование шрифтов (программа)

	#include<graphics.h>
	#include<conio.h>
	#include<stdio.h>
	#include<stdlib.h>
	#include"..\\h\\init.h"
	void main()
	{
		init();
		struct textsettingstype oldtext;
		int style; 

		// установить горизонтальное направление и выравнивание
		settextjustify(LEFT_TEXT, TOP_TEXT);
		// готический шрифт, масштабный коэффициент 7
		settextstyle(GOTHIC_FONT, HORIZ_DIR, 7);
		outtext("Gothic text");
		gettextsettings(&oldtext);	// получить текущие параметры
		getch();
		// установить другой шрифт и новые параметры для вывода нового текста
		settextjustify(0, 2 );
		style = installuserfont("arus.chr");

		//	 Можно поэкспериментировать со шрифтами
		//	style = installuserfont("bold.chr");
		//	style = installuserfont("euro.chr");
		//	style = installuserfont("simp.chr");
		//	style = installuserfont("trip.chr");
		//	style = installuserfont("sans.chr");
		//	style = installuserfont("scri.chr");
		//	style = installuserfont("goth.chr");

		settextstyle(style, 1, 7);
		outtextxy(20, 100, "Россия"); outtextxy(130, 250, "USA");
		getch();
		// опять новый шрифт и новые параметры для нового текста
		settextstyle(installuserfont("sans.chr"), 0, 17);
		outtextxy(150, 100, "Borland");
		getch();
		// восстановить полученные текущие параметры
		settextjustify(oldtext.horiz, oldtext.vert);
		settextstyle(oldtext.font, oldtext.direction, oldtext.charsize);
		outtextxy(350, 300, "End");
		getch();
		closegraph();
}
