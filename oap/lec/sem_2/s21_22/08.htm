Сортировка строк в алфавитном порядке

Рассмотрим программу сортировки строк в алфавитном порядке. Последовательность действий будет состоять из считывания массива строк, их сортировки и вывода. 
   #include<stdio.h>
   #include<string.h>
   #include<conio.h>
   stsrt(char *strings[], int num);
   #define SIZE   81			// предельная длина строки, включая \0
   #define LIM   20			// максимальное количество считываемых строк
   #define HALT ""			// нулевая строка для прекращения ввода
   main()
   {
	static char in[LIM][SIZE];	// массив вводимых строк
	char *ptstr[LIM];		// массив переменных типа указатель
	int ct = 0;			// счетчик вводимых строк
	int k;			// счетчик выводимых строк
	clrscr();
	printf("Введите до %d строк, и я их отсортирую.\n", LIM);
	printf("Для прекращения ввода - ENTER в начале строки.\n");
	while((gets(in[ct]) != NULL) && strcmp(in[ct], HALT) != 0 && ct++ < LIM)
		ptstr[ct-1] = in[ct-1];	// указывает на еще не отсортированный ввод
	stsrt(ptstr, ct);			// сортировка строк
	puts("\nОтсортированный список строк: \n");
	for (k = 0; k < ct; k++)
		puts(ptstr[k]);		// указатели на отсортированные строки
	getch(); return 0;
   }

   // функция сортировки строк с использованием указателей
   stsrt(char *strings[], int num)
   {
	char *temp;
	int top, seek;
	for (top = 0; top < num-1; top++)
		for(seek = top+1; seek < num; seek++)
			if (strcmp(strings[top], strings[seek]) > 0)
			{
				temp = strings[top];
				strings[top] = strings[seek];
				strings[seek] = temp;
			} return 0;
   }
