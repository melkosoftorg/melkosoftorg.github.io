Вложенные ветвления в цикле for

Рассмотрим программу, которая выводит прямоугольник из символов & размером 10 · 10.
i - управляющая переменная цикла - количество выводимых символов - 100.
j и k - переменные, задающие позицию вывода символа на экран.
Проверка условия необходима, чтобы после вывода 10 символов в строке перейти в начало следующей строки для вывода следующих 10 символов. И так далее все 10 строк по 10 символов. 

#include<conio.h>
#include<iostream.h>
main()
{
	int i, j = 1, k = 1,
	textcolor(12); clrscr();
	for(i = 1; i < 101; i++)
	{
		gotoxy(j++, k); cout << "&";
		if(i % 10 == 0)
		{
			k++; j = 1;
		}
	} getch(); return 0;
}

Используя оператор условного перехода if ... else, вычислить для значений x, изменяющихся от 0,9 до 3,1 с шагом 0,3, значения функции z.
Указать номер формулы, по которой производилось вычисление.
,   где    .

Управлять циклом будет переменная x, она изменяется строго закономерным образом от a до b с шагом h.
Проверять условие при каждом x после вычисления y необходимо для того, чтобы выбрать, по какой из формул вычислять z: либо по первой, либо по второй, либо по третьей. 

#include<conio.h>
#include<math.h>
#include<iostream.h>
#include<iomanip.h>
main()
{
	float y, x, z; int num; clrscr();
	cout.precision(2);
	for(x = .9; x <= 3.1; x += .3)
	{
		y = 2 * sin(M_PI_2 * x) + cos(sqrt(x));
		if(y < 0)
		{
			num = 1; z = exp(y) + exp(-y);
		}
		else	if(y >= 0 && y < 2)
			{
				num = 2; z = y * y + sqrt(y);
			}
			else
			{
				num = 3; z = y + log(y);
			}
		cout << setw(3) << "x = " << setw(3) << x;
		cout << setw(3) << "y = " << setw(5) << y << setw(3) << "z" << num << " = " << z << endl;
	} getch(); return 0;
}

Построить изображение круга, закрашенного чередующимися синими и красными кольцами.

Построение чередующихся синих и красных кругов начинается с самого большого синего, затем сверху меньшим радиусом рисуется красный круг с такими же координатами центра круга, что и у синего. И так далее все 7 кругов. В результате создается иллюзия полосатого круга.
При проверке условия не указана никакая операция отношения, просто записано (i % 2). Это означает истина, то есть имеется остаток от деления на 2, значение i - число нечетное. 

#include<graphics.h>
#include<conio.h>
#include<stdio.h>
#include<stdlib.h>
#include<dos.h>
#include"..\\h\\init.h"
main()
{
	init();
	int i, x = getmaxx() / 2, y = getmaxy() / 2, color;
	for(i = 7; i >= 1; i--)
	{
		if(i % 2)
			color = BLUE;
		else
			color = RED;
		setcolor(color); setfillstyle(SOLID_FILL, color);
		fillellipse(x, y, 20 * i, 20 * i);
		delay(300);
	}
	getch(); closegraph();
}
