Найти запись в файле

FINDT.CPP позволяет найти по введенной фамилии заданную запись в файле TEL_NUM.TXT и вывести на экран фамилию и соответствующий номер телефона. 

	// FINDT.CPP - позволяет найти строку str в файле TEL_NUM.TXT
	void Find(char *str)
	{	int i;
		// если файл невозможно открыть для чтения, то завершение работы
		if((F_tel = fopen(File, "r")) == NULL)
		{	fprintf(stderr, "\"%s\" невозможно открыть\n",File);
			exit(1);
		}
		// чтение числа записей (Count) в файле
		if(fread(&Count, sizeof(int), 1, F_tel) != 1)
		{	fprintf(stderr, "\"%s\" : ошибка чтения\n", File) ;
			exit(1);
		}
		// в цикле for осуществляется поиск нужной записи
		for(i = 0; i < Count; i++)
		{	fread(Name, 1, MAX_NAME, F_tel);		// читается имя
			fread(Number, 1, MAX_NUMBER, F_tel);	// читается номер
			// проверяется отсутствие ошибки
			if(ferror(F_tel))
			{	fprintf(stderr, "\"%s\"; ошибка чтения\n", File);
				exit(1);
			}
			// сравнивается с нужной строкой и если результат положительный, 
			// то фамилия и найденный номер телефона выводятся на экран
			if(strcmp(str, Name) == 0)
			{	printf("Фамилия: %s\n", Name);
				printf("Номер телефона: %s\n", Number);
				fclose(F_tel); return;
			}
		}
		// если результат поиска отрицательный, то выводится сообщение
		fprintf(stderr, "\"%s\": запись отсутствует в базе данных\n", File);
		fclose(F_tel); getch(); return;
	}
