Вывести базу данных на экран

SHOWT.CPP позволяет вывести на экран содержимое телефонного справочника. 

	// SHOWT.CPP - позволяет вывести все записи из файла TEL_NUM.TXT
	#include<conio.h>
	#include "myh.h"
	void Show(void)
	{	int i;
		// если файл невозможно открыть для чтения - завершение работы программы
		if((F_tel = fopen(File, "r")) == NULL)
		{	fprintf(stderr, "\"%s\" невозможно открыть\n", File);
			exit(1);
		}
		// чтение числа записей  Count  в файле
		if(fread(&Count, sizeof(int), 1, F_tel) != 1)
		{	fprintf(stderr, "\"%s\" : ошибка чтения\n", File);
			exit(1);
		}
		// в цикле for осуществляется вывод всех записей
		for(i = 0; i < Count; i++)
		{	fread(Name,   MAX_NAME,   1, F_tel);		// читается имя
			fread(Number, MAX_NUMBER, 1, F_tel);	// читается номер
			if(ferror(F_tel))			// проверяется отсутствие ошибки
			{	fprintf(stderr, "\"%s\": ошибка чтения\n", File);
				exit(1);
			}
			// вывод на экран фамилии и номера телефона
			printf("Фамилия: %s,  номер телефона: %s\n", Name, Number);
		} fclose(F_tel);
	}
	void main(void)
	{	clrscr();
		Show(); getch();
	}
