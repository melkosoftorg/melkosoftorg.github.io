Операции с элементами массива структур

С элементами массива структур можно производить различные операции. Например, средняя стоимость книги может быть вычислена: 
	sum = 0;
	for(i = 0; i < num; i++)
		sum += lib[i].val;
	cout << "Средняя цена книги = " << sum / num << endl;

В следующей программе создается массив структур, характеризующий массив N материальных точек, каждая из которых задана массой mi и тремя координатами (xi, yi, zi).
Вычисление координат центра масс производится по формулам


где общая масса системы точек равна 
	#include<conio.h>
	#include<stdio.h>
	// определение структурного типа
	struct part
	{
		double mass; 
		float coord[3];
	};
	void main()
	{	clrscr();
		// определение и инициализация массива структур
		struct part mass_point[] =	{20.0, {2.0, 4.0, 6.0}, 
					40.0, {6.0, -2.0, 8.0}, 
					10.0, {1.0, 3.0, 2.0} };
		int N;		// число точек
		// структура для формирования результата - центр масс
		struct part center = {0.0, {0.0, 0.0, 0.0}};
		int i, j;
		N = sizeof(mass_point) / sizeof(mass_point[0]);
		for(i = 0; i < N; i++)
		{
		   center.mass += mass_point[i].mass;
		   for(j = 0; j < 3; j++)
			center.coord[j] += mass_point[i].coord[j] * mass_point[i].mass;
		}
		printf("Координаты центра масс:");
		for(j = 0; j < 3; j++)
		{
			center.coord[j] /= center.mass;
			printf("\n %d\t%f", (j+1), center.coord[j]);
		}
		getch();
	}
