Генерация случайных чисел

В программе использованы различные функции для генерации случайных чисел и показаны возможные результаты. 
	#include<iostream.h>
	#include<iomanip.h>
	#include<conio.h>
	#include<stdlib.h>
	int main()
	{
		const int N = 10;
		int i, x[N], y[N];
		clrscr(); randomize();
		cout << "Случайные числа, диапазон 0 - 99\n" ;
		for (i = 0; i < N; i++)
		{
			x[i] = rand() % 100; y[i] = random(100);
			gotoxy(8 + i * 3, 2); cout << x[i];
			gotoxy(8 + i * 3, 3); cout << y[i];
		}
		cout << "\nСлучайные числа положительные и отрицательные\n";
		for (i = 0; i < N; i++)
		{
			x[i] = random(40) - random(70);
			gotoxy(8 + i * 4, 5); cout << x[i];
		}
		cout << "\nСлучайные вещественные числа\n";
		float z[N];
		cout.precision(2);
		for (i = 0; i < N; i++)
		{
			z[i] = (random(40) - random(70)) * 1.11;
			gotoxy(8 + i * 7, 7); cout << z[i];
		}
		getch(); return 0;
	}
