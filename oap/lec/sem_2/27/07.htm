Использование указателей для получения адреса

В следующей программе показано, как можно получить значения и адреса элементов массива и использование указателей. 
	#include<stdio.h>
	#include<conio.h>
	float x[] = {11.0, 25.0, 37.0, 44.0, 59.0, 66.0};
	main()  	
	{
		float *u1, *u2; int i; clrscr();
		printf("sizeof(float) = %d\n\n",sizeof(float));

		printf("Адреса указателей: &u1 = %p &u2 = %p\n", &u1, &u2);

		printf("\nАдреса элементов массива\n");
		for (i = 0; i < 6; i++)
		{
			if (i == 3)
				printf("\n");
			printf("&x[%d] = %p", i, &x[i]);
		}
		
		printf("\n\nЗначения элементов массива:\n");
		for (i = 0; i < 6; i++)
		{
			if (i == 3)
				printf("\n");
			printf("   x[%d] = %5.1f", i, x[i]);
		} printf("\n");
		
		for (u1 = &x[0], u2 = &x[5]; u2 >= &x[0]; u1++, u2--)
		{
			printf("\nu1 = %p *u1 = %5.1f u2 = %p *u2 = %5.1f", 
				u1, *u1, u2, *u2);
		}
		getch(); return 0;
	}
