Подсчет количества значений

Бросание шестигранной кости 500 раз и подсчет количества выпадения значений 1, 2, 3, 4, 5, 6. 
	#include<iostream.h>
	#include<iomanip.h>
	#include<conio.h>
	#include<stdlib.h>
	#include<time.h>
	main()
	{
		const int kol = 7;
		int num, i, x[kol] = {0};
		srand(time(NULL)); clrscr();
		// сделать 500 бросков
		for (i = 1; i <= 500; i++)
		{
			num = rand() % 6 + 1;		// выпавший номер от 1 до 6
			++x[num];				// подсчет количества выпадений
		}
		cout << "Значение" << setw(13) << "Количество\n";
		for (num = 1; num < kol; num++)
			cout << setw(4) << num << setw(10) << x[num] << endl;
		getch();return 0;
	}

	Результаты:

			Значение Количество
			1       	 99
			2       	 93
			3       	 77
			4       	 79
			5       	 80
			6        	 72

Следует обратить внимание, что при большом количестве повторов частота выпадения значений почти одинакова, то есть равновероятно получение любого из значений.
