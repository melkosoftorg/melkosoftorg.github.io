Матричное умножение двух двумерных массивов

Рассмотрим программу, которая определяет два двумерных массива
a[3][4]     и     b[4][2],     затем выполняет их матричное умножение: 
	#include<stdio.h>
	#include<conio.h>
	main()
	{
		int a[3][4] = { { 5,    3, -21, 42}, 
			       {44, 15,   0,   6}, 
			       {97,   6,  81,   2} };
		int b[4][2] = { {22,    7}, 
			       {97, -53}, 
			       {45,    0}, 
			       {72,    1} };
		int c[3][2], i, j, k; clrscr();
		printf("Вывод массива a[3][4]\n");
		for (i = 0; i < 3; i++)
		{
			for (j = 0; j < 4; j++)
				printf("%5d", a[i][j]);
			printf("\n");
		} 
		printf("\nВывод массива b[4][2]\n");
		for (i = 0; i < 4; i++)
		{
			for (j = 0; j < 2; j++)
				printf("%5d", b[i][j]);
			printf("\n");
		}
		for (i = 0; i < 3; i++) // матричное умножение
		{
			for (j = 0; j < 2; j++)
			{
				c[i][j] = 0;
				for (k = 0; k < 4; k++)
					c[i][j] += a[i][k] * b[k][j];
			}
		} 
		printf("\nВывод массива c[3][2]\n");
		for (i = 0; i < 3; i++)
		{
			for (j = 0; j < 2; j++)
				printf("%5d", c[i][j]);
			printf("\n");
		} getch(); return 0;
	}
