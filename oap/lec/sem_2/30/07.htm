Рассмотрим, каким образом можно осуществить передачу указателя на многомерный массив в функцию.

У нас имеется двумерный массив int x[S][K];
Объявим переменную-указатель, которая может указывать на элементы массива x, то есть на массивы из K целых значений: int (*ptr) [K];
Сделаем так, чтобы ptr указывал на первый элемент массива x:  ptr = x;

Внимание: круглые скобки в объявлении указателя на массив обязательны.
Квадратные скобки [ ] имеют более высокий приоритет, чем операция ссылки *. Поэтому оператор int *ptr[K]; объявляет массив из K указателей на значения типа int, а не указатель на массив.

В программе многомерный массив передается в функцию двумя способами - функции print_1( ) и print_2( ). 

	#include<stdio.h>
	#include<stdlib.h>
	#include<conio.h>
	const int S = 3, K = 4;
	// вывод элементов массива из K целых чисел
	void print_1(int (*ptr)[K])
	{	int *p, i;
		p = (int *) ptr;
		for(i = 0; i < K; i++)
			printf("%5d", *p++);
	}
	// вывод элементов n массивов по K чисел в каждом
	void print_2(int (*ptr)[K], int n)
	{	int *p, i;
		// p указывает на переменную типа int.
		// необходимо приведение типа для присвоения
		// указателю p адреса из ptr.
		p = (int *) ptr;
		for(i = 0; i < (K * n); i++)
			printf("%4d", *p++);
	}
	void main(void)
	{
		int i, j, x[S][K], (*ptr)[K];
		// Установить ptr - указатель на массив из K чисел типа int
		// на первый элемент массива x
		ptr = x;
		clrscr(); randomize();
		printf("\nСформирован массив:\n");
		for(i = 0; i < S; i++, printf("\n"))
			for(j = 0; j < K; j++)
				printf("%5d", (x[i][j] = rand() % 31));

		// При каждом проходе цикла ptr указывает на очередной элемент
		// массива x, то есть на очередной K-элементный массив целых чисел
		printf("\nprint_1()\n");
		for(i = 0; i < S; i++, printf("\n"))
		{	print_1(ptr++);
			getch();
		}
		printf("\nprint_2()\n");
		print_2(x, S);
		getch();
	}

			print_2()

Преобразование типов в команде присваивания p = (int *) ptr; необходимо, поскольку p и ptr являются указателями на разные типы данных:

  • p - указывает на int,
  • ptr - указывает на массив из K чисел типа int.
