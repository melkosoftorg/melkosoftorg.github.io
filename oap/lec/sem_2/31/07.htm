Библиотека собственных функций для работы с массивом

Если часто приходится работать с двумерным массивом, желательно пополнить библиотеку собственных функций функциями, относящимися к данному разделу. В следующие функции в качестве параметров необходимо передать массив и количество строк.

  • Например функция - сформировать массив (файл 20_mas.h) 
            void ff(int ls[][K], int s1)
        {
                int i, j;
                for (i = 0; i < s1; i++)
                        for(j = 0; j < K; j++) 
                                ls[i][j] = rand() % z;
        }

  • Или функция - вывести массив (файл 20_print.h) 
            void ww(int lss[][K], int s1)
        {
                int i, j;
                for(i = 0; i < s1; i++)
                {
                        k++;
                        for(j = 0; j < K; j++) 
                        {
                                gotoxy((j + 1) * 5, k); 
                                printf("%d ", lss[i][j]);
                        }
                } k++;
        }

В дальнейшем можно просто обращаться к этим функциям, например, как показано ниже, при создании одномерного массива p[K] максимальных значений каждого столбца. 

        #include<conio.h>
        #include<stdio.h>
        #include<stdlib.h>
        #define S 5	// количество строк
        #define K 4	// количество столбцов
        int z, k = 0;
	#include"..\\h\\20_print.h"
	#include"..\\h\\20_mas.h"
        int main()
        {
		clrscr(); randomize();
                int i, j, mas[S][K], p[K];

                // формирование и вывод массива
                z = 60; ff(mas, S); ww(mas, S);
 
                // Поиск максимального значения в столбце
                for(i = 0; i < K; i++)
                {
                        p[i] = mas[0][i];
                        for(j = 0; j < S; j++)
                                if(mas[j][i] > p[i]) 
                                        p[i] = mas[j][i];
                        gotoxy((j + 1) * 5, k + 2); printf("%d", p[i]);
                }
		getch(); return 0;
        }
