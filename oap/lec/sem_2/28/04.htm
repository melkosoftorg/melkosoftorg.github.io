Массивы динамической памяти

В программе показано, как можно ввести и напечатать в обратном порядке набор вещественных чисел, количество которых заранее не задано.

Для ввода - вывода значений использованы возможности C и C++. 

	#include<iostream.h>
	#include<stdio.h>	
	#include<conio.h>
	#include<stdlib.h>
	main()
	{  
		float *t,  mas;		// Указатель для выделяемого блока памяти
		int i, n;   clrscr();
		// ввод значений с клавиатуры
		printf("Количество элементов массива  n = "); scanf("%d", &n);
		/* Указатель на начало области, выделяемой для размещения
		   n  вводимых чисел. Принимает значение адреса области,
		   выделяемой для   n   штук значений типа  float
		*/
		t = (float*) malloc(n * sizeof(float));
		for(i = 0; i < n; i++)
		{
		  /* стандартный поток вывода  cout   в сочетании
		     с   << - оператором вставки - служит для отображения
		     литерал, значений констант и переменных без использования
		     указателей формата */
			cout << "t[" << i << "] = ";
		  /* стандартный поток ввода  cin   в сочетании
		     с   >> - оператором извлечения - служит для
		     считывания данных с клавиатуры */
			cin >> t[i];
		}
		// вывод элементов массива в обратном порядке
		for(i = n-1; i >= 0; i--)
		{
			if (i % 2 == 0) printf("\n");
				printf("\tt[%d] = %5.1f", i, t[i]);
		}
		// вывод адреса начального байта и значения элемента t[0]
		printf("\n\n\t&t = %p\t*t = %5.1f\n\n", &t, *t);
		free(t);		//  Освобождение памяти
		getch();
	}
