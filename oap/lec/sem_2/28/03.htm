Обращение к функциям при работе с массивом

Имея библиотеку функций, можно инициализировать массивы случайными числами, выводить их значения на экран, проводить сортировку массивов. 
	#include<conio.h>
	#include<stdio.h>
	#include<stdlib.h>
	#include<time.h>
	int k, num;   // описание глобальных переменных
	// описание функций собственной библиотеки
	void swap(int *a, int *b);
	void setrand(int list[], int size);
	void setrandom(int list[], int size);
	void get_mas(int list[], int size);
	main()
	{
		int i, j, y[10], z[10]; clrscr(); randomize();
		gotoxy(25,1); puts("Массивы \n");

		// заполнение массива y[10] значениями 0 - 999
		k = 1000; num = 10; setrandom(y, num);
		printf(" y[%d] с элементами 0 - %d\n\n", num, k-1); 
		get_mas(y, num);	// вывод массива y[10]

		// заполнение массива z[10] значениями 0 - 99 
		k = 100; num = 10; setrand(z, num);
		printf(" z[%d] с элементами 0 - %d\n\n", num, k-1); 
		get_mas(z, num);	// вывод массива z[10]

		// Сортировка - упорядочить по возрастанию
		printf(" z[10] по возрастанию\n\n");
		for(j = 0; j < 9; j++)
			for(i = j + 1; i < 10; i++)
				if(z[i] < z[j])
					swap(&z[i], &z[j]);
		get_mas(z, num);

		// Сортировка - упорядочить по убыванию
		printf(" z[10] по убыванию\n\n");
		for(j = 0; j < 9; j++)
			for(i = j + 1; i < 10; i++)
				if(z[i] > z[j]) 
					swap(&z[i], &z[j]);
		get_mas(z, num); getch(); return 0;
	}
