Посчитаем количество различных символов

Используя оператор множественного выбора switch ... case ... default ... break, подсчитать, сколько раз каждый из символов "Вет втр" встретился в фразе "Веет ветер".
Если введенного символа нет в перечне, вывести * и количество таких символов.

Счетчик количества символов организуем в массиве mas[10], где в качестве индекса будет выступать переменная p - номер присваиваемый определенному символу.
Значение каждого символа будем хранить в массиве str[p]. 


#include<conio.h>
#include<stdlib.h>
#include<stdio.h>
#include<iostream.h>
#include<iomanip.h>
main()
{  char ch, *str; int i, p; 
   int mas[10] = {0}; // инициализация массива нулями
   clrscr();
   while((ch = getchar()) != '\n')
      {   switch(ch)
           {  case 'В': p = 0; break;
              case 'в': p = 1; break;
              case ' ': p = 2; break; 
              case 'е': p = 3; break;
              case 'т': p = 4; break;  
              case 'р': p = 5; break;
              default : p = 6; ch = '*'; break; 
           }  ++mas[p]; str[p] = ch;
      }
   for(i = 0; i <= 6; i++)   
         cout << setw(1) << str[i] << setw(4) << mas[i] << endl;
   getch(); return 0;
}
