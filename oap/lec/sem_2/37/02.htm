Чтение структуры из файла

В приведенном ниже фрагменте программы показано, как можно из созданного на предыдущей странице файла proba.dat вычитать и вывести на экран массив структур. (Не забудьте вначале подключить необходимые библиотеки).

Например, функция читать массив структур из файла может иметь вид: 

	int read_struct(struct book New[], char *str)
	{
		FILE  *fp;
		int num = 0;
		// Открыть файл для чтения
		if ((fp = fopen(str, "r")) == NULL)
		{	puts("Невозможно открыть файл");
			exit(1);
		}
		while(fread(&New[num], sizeof(New[num]), 1, fp) == 1)
			num++;
		fclose(fp);
		return num;
	}

Объявлено три массива структур типа book:
  • lib[MAXB] - полный список (количество num),
  • frau[MAXB] - список женщин (количество nf),
  • man[MAXB] - список мужчин (количество nm).
	#define MAXT  40		// длина текста Ф.И.О.
	#define MAXA  40		// длина текста пол
	#define MAXB  15		// длина списка
	struct book
	{	// создание шаблона типа  book
		char tit[MAXT];
		char aut[MAXA];
		int val, god;
	};
	void main()
	{
		struct book lib[MAXB], frau[MAXB], man[MAXB];
		char *name = "proba.dat";
		int col;
		col = read_struct(lib, name);
		clrscr();
		print_struct(lib, col);
		getch();
	}
