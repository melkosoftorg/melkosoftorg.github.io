Преобразование строки
в длинное беззнаковое целое значение

strtoul преобразует строку в длинное беззнаковое целое значение по заданному основанию base, возвращает беззнаковое целое значение преобразованной строки или 0 в случае ошибки. Аналогична strtol за исключением того, что она преобразует строку в беззнаковое длинное целое значение.

Синтаксис unsigned long strtoul(char *str, char **endptr, int base);
Прототип в stdlib.h 

	#include<stdio.h>
	#include<stdlib.h>
	#include<conio.h>
	void main()
	{	char inp[80], *etr, *str = "87654321";
		long lnum; unsigned long unum;
		double val; clrscr();
		puts("Число с плавающей точкой: ");
		gets(inp);
		puts("strtod"); val = strtod(inp, &etr);
		printf(" Строка: %s, число: %lf\n", inp, val);
		puts("strtol"); lnum = strtol(str, &etr, 10);
		printf(" Строка: %s, число: %ld\n", str, lnum);
		puts("stroul"); unum = strtoul(str, &etr, 10);
		printf(" Строка: %s, число: %lu\n", str, unum);
		getch();
	}
