Построчная запись данных в файл
и чтение из файла

Программа демонстрирует возможности построчной записи данных в файл и чтение из файла и определение позиции указателя текущей позиции файла. 

	#include<stdio.h>
	#include<conio.h>
	#include<stdlib.h>
	#include<dir.h>
	#include<string.h>
	void main()
	{
		FILE *fp;
		char *name = "mim.txt";
		char str[55][80], buf[55][80];
		int j = 0; clrscr();
		puts("Ввести данные в файл");
		if ((fp = fopen(name, "w")) == NULL)		// Открыть файл для записи
		{ 
			puts("Невозможно открыть файл"); exit(1); 
		}
		printf("%s\n\n", searchpath(name));		// Маршрут доступа к файлу
		printf("1) Положение указателя текущей позиции файла = %d\n", ftell(fp));
		while(strlen(gets(str[j])) > 0)
		{
			fputs(str[j], fp); fputc('\n', fp);
			j++;
		}
		printf("2) Положение указателя текущей позиции файла = %d\n", ftell(fp));
		fclose(fp);				// Закрыть поток
		fp = fopen(name, "r");			// Открыть файл для чтения
		if (fp == NULL)
		{ 
			puts("Невозможно открыть файл"); exit(1); 
		}
		printf("3) Положение указателя текущей позиции файла = %d\n", ftell(fp));
		j = 0;
		fseek(fp, 0, SEEK_SET);			// Перейти в начало файла
		printf("4) Положение указателя текущей позиции файла = %d\n", ftell(fp));
		while(fgets(buf[j], strlen(str[j]) + 1, fp))
		{ 
			printf("%s",buf[j]); j++; 
		}
		printf("str[%d][%d] = %c\n", 3, 5, str[3][5]); 
		printf("str[%d] = %s\n", 3, str[3]);
		printf("5) Положение указателя текущей позиции файла = %d\n", ftell(fp));
		fclose(fp); getch();
	}
