Ѕесконечный цикл

Ѕесконечный цикл - это цикл, выполнение которого никогда не закончитс€ естественным образом по его условию.

¬ следующей программе показано, как можно использовать цикл for при построении движущегос€ изображени€, создать бесконечный цикл, прерываемый командой break при нажатии на любую клавишу. 

	#include<graphics.h>
	#include<stdlib.h>
	#include<stdio.h>
	#include<math.h>
	#include<conio.h>
	#include<dos.h>
	#include"..\\h\\init.h"
	void main()
	{
		init(); randomize();
		int color, x, y, x1, y1, dy, r, h, w;
		float i;
		x1 = 250; y1 = 30; w = 350; h = 410;
		setbkcolor(0); cleardevice();
		setfillstyle(1, 11); bar(x1 - 8, y1 - 8, x1 + w + 8, y1 + h + 8);
		setfillstyle(1, 9); bar(x1 - 4, y1 - 4, x1 + w + 4, y1 + h + 4);
		// создать графическое окно и рисовать в нем салют
		setviewport(x1, y1, x1 + w, y1 + h, 1); clearviewport();
		x = w / 2;  dy = h / 50;
		setcolor(7); setfillstyle(1, 7); r = 1;
		// снар€д летит вверх
		for(y = h - dy; y > 2 * h / 3; y -= dy)
		{
			setcolor(7); setfillstyle(1, 7); fillellipse(x, y, r, r);
			delay(90);
			setcolor(0); setfillstyle(1, 0); fillellipse(x, y, r, r);
			if(y % 3 == 0)
				r++;
		}
		setcolor(2); setfillstyle(1, 2); fillellipse(x, y, r, r);
		delay(70);
		setcolor(0); setfillstyle(1, 0); fillellipse(x, y, r, r);
		r = 1;
		// бесконечный цикл - салют
		for(;;)
		{
			color = 1 + random(15); setcolor(color);
			i = random(280) * .011; setfillstyle(1, color);
			x1 = x + random(w / 3) * cos(i); y1 = y - random(h / 2) * sin(i);
			r = 1 + random(2); fillellipse(x1, y1, r, r);
			delay(70);
			setcolor(0); setfillstyle(1, 0); fillellipse(x1, y1, r, r);
			for(i = 0; i <= 2 * M_PI; i += M_PI_4 / 5)
			{
				putpixel(x1 + random(30) * cos(i), y1 + random(20) * sin(i), color);
				delay(1);
			}
			delay(50);
			// выйти из цикла на нажатие любой клавиши
			if(kbhit())
				break;
		}
		getch(); closegraph();
	}
