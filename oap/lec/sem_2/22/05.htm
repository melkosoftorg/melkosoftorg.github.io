Операторы break и continue

Если необходимо выйти из цикла до его завершения используется оператор break.

Выполнение оператора continue приводит к пропуску оставшихся операторов в теле цикла и выполнению следующей итерации цикла.

Программа Результаты

#include<conio.h>
#include<iostream.h>
main()
{	int i; clrscr();

	cout << "Оператор break\n";
	for(i = 1; i <= 10; i++)
	{	if(i == 6)
			break;		// прерывает цикл при i = 6
		cout << i << ' ';
	}

	cout << "\n\nОператор continue\n";
	for(i = 1; i <= 10; i++)
	{	if(i == 6)
			continue;	// пропускает код в цикле при i = 6
		cout << i << ' ';
	} getch(); return 0;
}

Оператор break:

  • можно использовать во всех трех циклах for, while и do... while,
  • можно использовать в конструкции switch,
  • нельзя использовать:
    • в конструкции if ... else, не вложенной в тело цикла,
    • в теле главной процедуры main для выхода из нее.

Программный код, реализующий эти же действия в циклах while и do... while.

Для цикла while Для цикла do...while 
cout << "Оператор break\n";
i = 0;
while(++i <= 10)
{
	if(i == 6)
		break;
	cout << i << ' ';
}

cout << "\n\nОператор continue\n";
i = 0;
while(++i <= 10)
{
	if(i == 6)
		continue;
	cout << i << ' ';
}

cout << "Оператор break\n";
i = 1;
do
{
	if(i == 6)
		break;
	cout << i << ' ';
} while(i++ < 10);

cout << "\n\nОператор continue\n";
i = 1;
do
{
	if(i == 6)
		continue;
	cout << i << ' ';
} while(i++ < 10);
