Размещения объединения в памяти

Заносить значения в участок памяти, выделенный для объединения, можно с помощью любого из его элементов.
То же самое справедливо и относительно доступа к содержимому участка памяти, выделенного для объединения. 
	#include<iostream.h>
	#include<conio.h>
	main()
	{
		union	// Определение объединения
		{	
			double l; 
			int i1, i2;
			char c[6];
		} uni = { 34.9 };
		clrscr();
		cout << " Размещения объединения в памяти\n\n";
		cout << "&uni = " << &uni << " sizeof(uni) = " << sizeof(uni);
		cout << "\n\n&uni.l = " << &uni.l << " sizeof(uni.l) = " << sizeof(uni.l);
		cout << "\n&uni.i1 = " << &uni.i1 << " sizeof(uni.i1) = "<< sizeof(uni.i1);
		cout << "\n&uni.i2 = " << &uni.i2 << " sizeof(uni.i2) = " << sizeof(uni.i2);
		cout << "\n&uni.c = " << &uni.c << " sizeof(uni.c) = " << sizeof(uni.c);
		getch();
	}

		Размещение объединения в памяти

		&uni = 0x8f4bffee sizeof(uni) = 8
		&uni.l = 0x8f4bffee sizeof(uni.l) = 8
		&uni.i1 = 0x8f4bffee sizeof(uni.i1) = 2
		&uni.i2 = 0x8f4bffee sizeof(uni.i2) = 2
		&uni.c = 0x8f4bffee sizeof(uni.c) = 6
Вывод: все элементы объединения имеют один и тот же начальный адрес, размеры элементов соответствуют их типам, размер объединения определяется максимальным размером его элементов.
