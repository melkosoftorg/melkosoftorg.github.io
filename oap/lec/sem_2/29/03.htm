Находим максимальное и минимальное значения функции

Для того, чтобы вычислить размеры графика необходимо знать значения
          xmin,     ymin,     xmax,     ymax.

Минимальное и максимальное значения аргумента x соответствуют концам отрезка [-1, 3].

Минимальное и максимальное значения функции, а также соответствующие им значения аргументов вычислим по программе 

	float ymax, ymin, x1, x2;
	int i;
	ymax = ymin = y[0];
	x1 = x2 = x[0];
	for(i = 1; i < num; i++)
	{
		if(y[i] > ymax)
		{
			ymax = y[i]; x1 = x[i];
		}
		if(y[i] < ymin)
		{
			ymin = y[i]; x2 = x[i];
		}
	}

Итак:

  • сформированы массивы значений аргументов x и функций y;
  • найдены максимальное и минимальное значения функции и соответствующие им значения аргументов.

Необходимо создать функции для вывода значений:

  • на экран: 
     
	// вывод на экран
	void print_mas(float a, float b, float ymin, float x2, float ymax, float x1)
	{
	        gotoxy(20, 5); cout << "На отрезке [" << a << ", " << b << "]";
	        gotoxy(20, 7); cout << "Ymin(" << x2 << ") = " << ymin;
	        gotoxy(20, 9); cout << "Ymax(" << x1 << ") = " << ymax << endl;
	}
  • в файл: 
     
	// записать в файл
	int fil_mas(float y[], float x[], int size, float a, float b, float ymin, float x2, float ymax, float x1)
	{
		int i; FILE *f; char  *ch = "..\\fil_dat\\pr21_iii.dat";
		if ((f = fopen(ch, "wt")) == NULL)
		{
			fprintf(stderr, "Ошибка! Не удается открыть файл...\n");
			return 1;
		}
		fprintf (f, "%f %f\n", a, b);
		fprintf (f, "%f %f\n", x2, ymin);
		fprintf (f, "%f %f\n", x1, ymax);
		fprintf (f, "%d\n", size);
		for(i=0;  i < size; i++)
		{
			fprintf (f, "%f %f\n", x[i], y[i]);
		}
		fclose (f); return 0;
	}

Осуществить вызов этих функций с соответствующими фактическими значениями аргументов 

 
	print_mas(a, b, ymin, x2, ymax, x1);
	fil_mas(y, x, num, a, b, ymin, x2, ymax, x1);

В результате получим на экране: 

 
	Вычисление значений функции y = x * x * sin(M_PI * x)
		На отрезке [-1, 3]
		Ymin(1.619999) = -2.440105
		Ymax(2.579998) = 6.447278

В файл помимо указанных значений будут выведены массивы аргументов и функций, а также количество элементов массивов.
