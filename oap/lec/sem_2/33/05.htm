Сравнение

strcmp сравнивает строки str1 и str2, начиная с первого символа в каждой строке и продолжая сравнение последующих символов до тех пор, пока не встретятся несовпадающие символы или строки не кончатся.
stricmp сравнивает строки str1 и str2 без различий регистров. 
Синтаксис	int strcmp(char *str1, char *str2);
		int stricmp(char *str1, char *str2);
Прототип в 	string.h

	#include<string.h>
	#include<stdio.h>
	#include<conio.h>
	main()
	{	char *buf1 = "babc", *buf2 = "babc", *buf3 = "BaBc";
		int ptr; clrscr();
		puts("strcmp");
		ptr = strcmp(buf2, buf1);
		if(ptr == 0)
			printf(" buf2: %s == buf1: %s\n", buf2, buf1);
		else 
			printf(" buf2: %s != buf1: %s\n",buf2, buf1);
		ptr = strcmp(buf2, buf3);
		if (ptr == 0) 
			printf(" buf2: %s == buf3: %s\n",buf2, buf3);
		else 
			printf(" buf2: %s != buf3: %s\n",buf2, buf3);
		puts("stricmp");
		ptr = stricmp(buf2, buf1);
		if(ptr == 0) 
			printf(" buf2: %s == buf1: %s\n",buf2, buf1);
		else 
			printf(" buf2: %s != buf1: %s\n",buf2, buf1);
		ptr = stricmp(buf2, buf3);
		if (ptr == 0) 
			printf(" buf2: %s == buf3: %s\n",buf2, buf3);
		else 
			printf(" buf2: %s != buf3: %s\n",buf2, buf3);
		getch(); return 0;
	}

	Результат	strcmp
			 buf2: babc == buf1: babc
			 buf2: babc != buf3: BaBc

			stricmp
			 buf2: babc == buf1: babc
			 buf2: babc == buf3: BaBc
strncmp сравнивает часть одной строки с частью другой, просматривает maxlen символов.
strnicmp сравнивает часть одной строки с другой без различий регистров на максимальной длине maxlen байт.
strncmpi сравнивает часть строки str1 с частью строки str2 без различий регистра на максимальной длине n байт. 
Синтаксис	int strncmp(char *stp1, char *stp2, int maxlen);
		int strnicmp(char *str1, char *str2, unsigned maxlen);
		int strncmpi(char *str1, char *str2, size_t n);
Прототип в	string.h
Функции strcmp, stricmp, strncmp, strnicmp, strncmpi возвращают значение:
  • < 0, если str1 меньше, чем str2,
  • == 0, если str1 то же самое, что str2,
  • > 0, если str1 больше, чем str2. 
	#include<string.h>
	#include<stdio.h>
	#include<conio.h>
	main()
	{	char *buf1 = "babec", *buf2 = "Babhc",*buf3 = "Babec";
		int ptr; clrscr();
		puts("strncmp(buf2, buf1, 3)");
		ptr = strncmp(buf2, buf1, 3);
		if(ptr > 0) 
			printf(" buf2: %s > buf1: %s\n",buf2, buf1);
		if(ptr < 0) 
			printf(" buf2: %s < buf1: %s\n",buf2, buf1);
		if(ptr == 0)
			printf(" buf2: %s == buf1: %s\n",buf2, buf1);
		puts("strncmp(buf2, buf3, 3)");
		ptr = strncmp(buf2, buf3, 3);
		if(ptr > 0) 
			printf(" buf2: %s > buf3: %s\n",buf2, buf3);
		if(ptr < 0) 
			printf(" buf2: %s < buf3: %s\n",buf2, buf3);
		if(ptr == 0) 
			printf(" buf2: %s == buf3: %s\n",buf2, buf3);
		puts("strncmp(buf1, buf3, 3)");
		ptr = strncmp(buf1, buf3, 3);
		if(ptr > 0) 
			printf(" buf1: %s > buf3: %s\n",buf1, buf3);
		if(ptr < 0) 
			printf(" buf1: %s < buf3: %s\n",buf1, buf3);
		if(ptr == 0)
			printf(" buf1: %s == buf3: %s\n",buf1, buf3);
		puts("strnicmp(buf2, buf1, 3)");
		ptr = strnicmp(buf2, buf1, 3);
		if(ptr > 0) 
			printf(" buf2: %s > buf1: %s\n",buf2, buf1);
		if(ptr < 0) 
			printf(" buf2: %s < buf1: %s\n",buf2, buf1);
		if(ptr == 0)
			printf(" buf2: %s == buf1: %s\n",buf2, buf1);
		puts("strncmpi(buf2, buf1, 3)");
		ptr = strncmpi(buf2, buf1, 3);
		if(ptr > 0) 
			printf(" buf2: %s > buf1: %s\n",buf2, buf1);
		if(ptr < 0) 
			printf(" buf2: %s < buf1: %s\n",buf2, buf1);
		if(ptr == 0)
			printf(" buf2: %s == buf1: %s\n",buf2, buf1);
		getch(); return 0;
	}
